function solution(words) {
  let answer;
  let N = words[0].length;
  let i; // answerë¡œ ë°˜í™˜í•˜ê¸° ìœ„í•´ forë¬¸ scope ë°–ì—ì„œ ì„ ì–¸í•´ ì¤Œ.
  let sh = new Map();

    for (let j = 0; j < words.length; j++) {
