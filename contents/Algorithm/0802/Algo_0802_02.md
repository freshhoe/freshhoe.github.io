## A. 풀이

```javascript
// 전위 순회
function solution1 (n) {
    let answer = '';
    function DFS (v) {
        if (v > 7)  {
            return;
        } else {
            answer += (v + ' ');
            DFS(v * 2); // 왼쪽 노드
            DFS(v * 2 + 1); // 오른쪽 노드
        }
    }
    DFS(n);
    return answer;
}
console.log(solution1(1));
```
